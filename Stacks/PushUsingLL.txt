public class Main { 
  
    StackNode root; 
  
    static class StackNode { 
        int data; 
        StackNode next; 
  
        StackNode(int data) 
        { 
            this.data = data; 
        } 
    } 
    public void push(int data) 
    { 
        StackNode newNode = new StackNode(data); 
  
        if (newNode== null) {
            System.out.println("stack overflow");
        } 
        else { 
            newNode.next = root; 
            root = newNode; 
        } 
        System.out.println(data + " pushed to stack"); 
    } 
  
    public static void main(String[] args) 
    { 
  
        Main sll = new Main(); 
   sll.push(10); 
        sll.push(20); 
        sll.push(30); 
       
    } 
}
